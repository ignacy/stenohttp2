def build
  cipher = OpenSSL::Cipher.new(CIPHER_TYPE)
  encryptor ? cipher.encrypt : cipher.decrypt
  cipher.key = OpenSSL::PKCS5.pbkdf2_hmac_sha1(
    DEFAULT_PASSWORD, DEFAULT_SALT, 20_000, cipher.key_len
  )
  cipher
end
