def start
  puts "Listening on #{path} for *.message"
  listener = Listen.to(path) do |modified, added, _|
    new_file = added&.first || modified&.first
    decode_message(File.read(new_file)) if new_file&.end_with?('message')
  end
  listener.start
  sleep
end
